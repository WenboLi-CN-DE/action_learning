<!-- FastAPI 应用入口模式（来自 ARCHITECTURE.md Pattern 3） -->
```python
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from app.api import health

app = FastAPI(title="AI工坊平台", version="0.1.0")

# CORS 配置（来自 PITFALLS.md Pitfall 8 — 防止 CORS 陷阱）
app.add_middleware(
    CORSMiddleware,
    allow_origins=["http://localhost:5173"],  # Vite dev server
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# API Router 挂载
api_router = APIRouter(prefix="/api/v1")
api_router.include_router(health.router, tags=["health"])
app.include_router(api_router)
```

<!-- 数据库配置模式（来自 ARCHITECTURE.md Pattern 2） -->
```python
from sqlmodel import Session, create_engine, SQLModel

sqlite_url = "sqlite:///./database.db"
connect_args = {"check_same_thread": False}  # SQLite 多线程必需
engine = create_engine(sqlite_url, echo=True, connect_args=connect_args)

def get_session():
    with Session(engine) as session:
        yield session

def create_db_and_tables():
    SQLModel.metadata.create_all(engine)
```
